
# https://docs.gitlab.com/runner/install/docker.html
#
# instead creating a volume, bind mount the config directory in the host.
# this allows to adjust the configuration.
# for instance set "pull_policy = if-not-present

docker run -d --name gitlab-runner --restart always -v /var/run/docker.sock:/var/run/docker.sock -v /etc/gitlab-runner:/etc/gitlab-runner gitlab/gitlab-runner:latest
